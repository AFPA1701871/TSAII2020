#include <stdio.h>
#include <stdlib.h>

int main()
{
     int i,total=0 ;
    int note[9];
	double moyenne;


    //Remplissage du tableau note
    for (i=0; i<9; i++)
    {
        printf("\nentrer une valeur");
        scanf("%d",&note[i]);
		total=total+note[i];
    }

    //Affichage du tableau note
    for(i=0; i<9; i++)
    {
        printf("%d | ",note[i]);
    }
	
	moyenne=total/9;
	printf("La moyenne est de %f",moyenne);
	
    return 0;
}
